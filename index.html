var init_count = 0;
var close_count = 0;
var close_complete_count = 0;
$(document).ready(function(){
    
    $('.alert-callback').bind('click', function(){
        metroAlert('Alert with callbacks.', {
            
            // set style
            theme: metroStyle.INFO,
            
            // trigger when initialize trigger
            onInit: function(){
                $('.alert-cb-init').html("Intialize " + ++init_count);
            },
            
            // trigger when close button press
            onCloseInit: function(){
                $('.alert-cb-close-init').html("Close Initialized " + ++close_count);
            },
            
            // trigger when compelte closed
            onCloseComplete: function(){
                $('.alert-cb-close-complete-init').html("Close Complete " + ++close_complete_count);
            }
        }); 
    });
});
                            
                        
                            
// alert with update ui
var delay_show_button = 5;   // 5 seconds
$(document).ready(function(){
    $('.alert-update').bind('click', function(){
        metroAlert('Please wait... ' + delay_show_button, {
            
            // set style
            theme: metroStyle.INFO,
            
            // hide close button
            showCloseButton: false
        });
        
        var interval = setInterval(function(){
            --delay_show_button;
            if(delay_show_button < 1){
                clearInterval(interval);
                metroAlert('Task Completed.', {
                    theme: metroStyle.SUCCESS,
                    updateMetro: true,
                    showCloseButton: true
                });
                
            }
            else{
                metroAlert('Please wait... ' + delay_show_button, {
                    theme: metroStyle.INFO,
                    updateMetro: true,
                    showCloseButton: false
                });
            }
        }, 1000);
    });
});
                            
                        
                            
// alert with auto close
var delay = 6;   // 6 seconds
$(document).ready(function(){                                
    $('.alert-auto-close').bind('click', function(){
        delay = 6;
        metroAlert('Please wait... ' + delay, {
            
            // set style
            theme: metroStyle.INFO,
            
            // auto close true
            autoClose: true,
            
            // set auto close time optional default is 5 seconds
            autoCloseTime: (delay + 1) * 1000,
            
            // hide close button
            showCloseButton: false
        });
                            
        var interval = setInterval(function(){
            --delay;
            if(delay < 1){
                clearInterval(interval);
                metroAlert('Task Completed.', {
                    theme: metroStyle.SUCCESS,
                    updateMetro: true,
                    showCloseButton: false
                });
                
            }
            else{
                metroAlert('Please wait... ' + delay, {
                    theme: metroStyle.INFO,
                    updateMetro: true,
                    showCloseButton: false
                });
            }
        }, 1000);
    });
});

                });
                
            }

                metroAlert('Please wait... ' + delay, {
                    theme: metroStyle.INFO,
                    updateMetro: true,
                    showCloseButton: false
                });
            }
        }, 1000);
    });
});
